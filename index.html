Basic Usage

// ES Modules
import { CountryFlags } from 'svg-country-flags-2x1';

// Create an instance
const flags = new CountryFlags();

// Get a flag URL
const flagUrl = flags.getFlagUrl('us');

// Get flag with details
const flagDetails = await flags.getFlag('fr', { width: 200 });
